**Goal:** Build a lightweight LXMF messenger on Reticulum with a Textual TUI — NomadNet stripped to just messaging.

**Architecture:** Single-process app that initializes Reticulum, runs an LXMF router, stores messages/contacts/announces in SQLite, and presents a Textual TUI. CLI mode (`fieldmsg send`) and daemon mode (`--daemon`) skip the TUI entirely. LXMF callbacks bridge into the Textual event loop via `call_from_thread`.

**Tech Stack:** Python 3.10+, `rns`, `lxmf`, `textual`, `sqlite3`, `tomli`/`tomllib`

```python
from __future__ import annotations

import os
import sys
from dataclasses import dataclass, field
from pathlib import Path

if sys.version_info >= (3, 11):
    import tomllib
else:
    import tomli as tomllib


DEFAULT_FIELDMSG_DIR = os.path.expanduser("~/.fieldmsg")


@dataclass
class Config:
    # [identity]
    display_name: str = "fieldmsg"

    # [lxmf]
    propagation_node: str = ""
    sync_interval: int = 300
    announce_at_start: bool = True
    announce_interval: int = 600

    # [storage]
    db_path: str = "~/.fieldmsg/messages.db"
    max_age_days: int = 90

    # runtime (not from file)
    config_path: str | None = None
    rns_config_dir: str | None = None

    @property
    def fieldmsg_dir(self) -> str:
        return DEFAULT_FIELDMSG_DIR

    @property
    def identity_path(self) -> str:
        return os.path.join(self.fieldmsg_dir, "identity")

    @property
    def storage_path(self) -> str:
        return os.path.join(self.fieldmsg_dir, "storage")

    def resolve_db_path(self) -> str:
        return os.path.expanduser(self.db_path)


def load_config(path: str | None = None) -> Config:
    cfg = Config()
    if path is None:
        path = os.path.join(DEFAULT_FIELDMSG_DIR, "config.toml")

    cfg.config_path = path

    if not os.path.isfile(path):
        return cfg

    with open(path, "rb") as f:
        data = tomllib.load(f)

    identity = data.get("identity", {})
    cfg.display_name = identity.get("display_name", cfg.display_name)

    lxmf = data.get("lxmf", {})
    cfg.propagation_node = lxmf.get("propagation_node", cfg.propagation_node)
    cfg.sync_interval = lxmf.get("sync_interval", cfg.sync_interval)
    cfg.announce_at_start = lxmf.get("announce_at_start", cfg.announce_at_start)
    cfg.announce_interval = lxmf.get("announce_interval", cfg.announce_interval)

    storage = data.get("storage", {})
    cfg.db_path = storage.get("db_path", cfg.db_path)
    cfg.max_age_days = storage.get("max_age_days", cfg.max_age_days)

    return cfg


def generate_example_config() -> str:
    return Path(__file__).resolve().parent.parent.parent.joinpath(
        "examples", "config.toml"
    ).read_text()
```

```python
from __future__ import annotations

import sqlite3
import time
from typing import Any


def _row_to_dict(cursor: sqlite3.Cursor, row: tuple) -> dict[str, Any]:
    return {col[0]: row[i] for i, col in enumerate(cursor.description)}


class Store:
    def __init__(self, db_path: str):
        self.db = sqlite3.connect(db_path, check_same_thread=False)
        self.db.row_factory = _row_to_dict
        self.db.execute("PRAGMA journal_mode=WAL")
        self._create_tables()

    def _create_tables(self) -> None:
        self.db.executescript("""
            CREATE TABLE IF NOT EXISTS messages (
                id TEXT PRIMARY KEY,
                peer_hash TEXT NOT NULL,
                direction TEXT NOT NULL,
                content TEXT NOT NULL,
                timestamp REAL NOT NULL,
                status TEXT DEFAULT 'pending',
                read INTEGER DEFAULT 0
            );

            CREATE TABLE IF NOT EXISTS contacts (
                hash TEXT PRIMARY KEY,
                nickname TEXT NOT NULL,
                display_name TEXT,
                last_seen REAL,
                trusted INTEGER DEFAULT 0
            );

            CREATE TABLE IF NOT EXISTS announces (
                hash TEXT NOT NULL,
                display_name TEXT,
                hops INTEGER,
                timestamp REAL NOT NULL,
                interface TEXT
            );

            CREATE INDEX IF NOT EXISTS idx_messages_peer
                ON messages(peer_hash, timestamp);
            CREATE INDEX IF NOT EXISTS idx_announces_time
                ON announces(timestamp);
        """)

    # --- Messages ---

    def save_message(
        self, msg_id: str, peer_hash: str, direction: str,
        content: str, timestamp: float, status: str,
    ) -> None:
        read = 0 if direction == "in" else 1
        self.db.execute(
            "INSERT OR REPLACE INTO messages (id, peer_hash, direction, content, timestamp, status, read) "
            "VALUES (?, ?, ?, ?, ?, ?, ?)",
            (msg_id, peer_hash, direction, content, timestamp, status, read),
        )
        self.db.commit()

    def get_messages(self, peer_hash: str, limit: int = 200) -> list[dict]:
        return self.db.execute(
            "SELECT * FROM messages WHERE peer_hash = ? ORDER BY timestamp ASC LIMIT ?",
            (peer_hash, limit),
        ).fetchall()

    def update_message_status(self, msg_id: str, status: str) -> None:
        self.db.execute(
            "UPDATE messages SET status = ? WHERE id = ?", (status, msg_id)
        )
        self.db.commit()

    def get_conversations(self) -> list[dict]:
        return self.db.execute("""
            SELECT m.peer_hash,
                   m.content AS last_message,
                   m.timestamp AS last_timestamp,
                   m.direction AS last_direction,
                   COALESCE(c.nickname, m.peer_hash) AS display_name,
                   (SELECT COUNT(*) FROM messages u
                    WHERE u.peer_hash = m.peer_hash AND u.read = 0) AS unread
            FROM messages m
            INNER JOIN (
                SELECT peer_hash, MAX(timestamp) AS max_ts
                FROM messages GROUP BY peer_hash
            ) latest ON m.peer_hash = latest.peer_hash AND m.timestamp = latest.max_ts
            LEFT JOIN contacts c ON m.peer_hash = c.hash
            ORDER BY m.timestamp DESC
        """).fetchall()

    def get_unread_count(self, peer_hash: str) -> int:
        row = self.db.execute(
            "SELECT COUNT(*) as cnt FROM messages WHERE peer_hash = ? AND read = 0",
            (peer_hash,),
        ).fetchone()
        return row["cnt"] if row else 0

    def mark_read(self, peer_hash: str) -> None:
        self.db.execute(
            "UPDATE messages SET read = 1 WHERE peer_hash = ? AND read = 0",
            (peer_hash,),
        )
        self.db.commit()

    def delete_old_messages(self, max_age_days: int) -> int:
        cutoff = time.time() - (max_age_days * 86400)
        cur = self.db.execute(
            "DELETE FROM messages WHERE timestamp < ?", (cutoff,)
        )
        self.db.commit()
        return cur.rowcount

    # --- Contacts ---

    def save_contact(
        self, hash: str, nickname: str,
        display_name: str | None = None,
        last_seen: float | None = None,
        trusted: int = 0,
    ) -> None:
        self.db.execute(
            "INSERT INTO contacts (hash, nickname, display_name, last_seen, trusted) "
            "VALUES (?, ?, ?, ?, ?) "
            "ON CONFLICT(hash) DO UPDATE SET nickname=?, display_name=COALESCE(?, display_name), "
            "last_seen=COALESCE(?, last_seen), trusted=?",
            (hash, nickname, display_name, last_seen, trusted,
             nickname, display_name, last_seen, trusted),
        )
        self.db.commit()

    def get_contact(self, hash: str) -> dict | None:
        return self.db.execute(
            "SELECT * FROM contacts WHERE hash = ?", (hash,)
        ).fetchone()

    def get_contacts(self) -> list[dict]:
        return self.db.execute(
            "SELECT * FROM contacts ORDER BY nickname"
        ).fetchall()

    def delete_contact(self, hash: str) -> None:
        self.db.execute("DELETE FROM contacts WHERE hash = ?", (hash,))
        self.db.commit()

    def find_contact_by_nickname(self, nickname: str) -> dict | None:
        return self.db.execute(
            "SELECT * FROM contacts WHERE LOWER(nickname) = LOWER(?)", (nickname,)
        ).fetchone()

    def update_contact_last_seen(self, hash: str, timestamp: float, display_name: str | None = None) -> None:
        if display_name:
            self.db.execute(
                "UPDATE contacts SET last_seen = ?, display_name = ? WHERE hash = ?",
                (timestamp, display_name, hash),
            )
        else:
            self.db.execute(
                "UPDATE contacts SET last_seen = ? WHERE hash = ?",
                (timestamp, hash),
            )
        self.db.commit()

    # --- Announces ---

    def save_announce(
        self, hash: str, display_name: str | None,
        hops: int, timestamp: float, interface: str | None,
    ) -> None:
        self.db.execute(
            "INSERT INTO announces (hash, display_name, hops, timestamp, interface) "
            "VALUES (?, ?, ?, ?, ?)",
            (hash, display_name, hops, timestamp, interface),
        )
        self.db.commit()

    def get_announces(self, limit: int = 100) -> list[dict]:
        return self.db.execute(
            "SELECT * FROM announces ORDER BY timestamp DESC LIMIT ?", (limit,)
        ).fetchall()

    def close(self) -> None:
        self.db.close()
```

```python
class AnnounceHandler:

class Core:
    """Reticulum + LXMF router management."""

    def __init__(self, config: Config):
        self.config = config
        self.reticulum = None
        self.identity = None
        self.lxmf_router = None
        self.local_destination = None
        self.store: Store | None = None

        # Callbacks — the TUI or CLI sets these
        self.on_message: Callable | None = None
        self.on_announce: Callable | None = None
        self.on_delivery_status: Callable | None = None

    @property
    def identity_path(self) -> str:
        return getattr(self.config, "_identity_path", self.config.identity_path)

    @property
    def storage_path(self) -> str:
        return getattr(self.config, "_storage_path", self.config.storage_path)

    @property
    def fieldmsg_dir(self) -> str:
        return getattr(self.config, "_fieldmsg_dir", self.config.fieldmsg_dir)

    def setup(self) -> None:
        """Initialize Reticulum, LXMF, identity, and store."""
        # Ensure directories exist
        os.makedirs(self.fieldmsg_dir, exist_ok=True)
        os.makedirs(self.storage_path, exist_ok=True)

        # Initialize store
        db_path = self.config.resolve_db_path()
        os.makedirs(os.path.dirname(db_path), exist_ok=True)
        self.store = Store(db_path)

        # Initialize Reticulum
        self.reticulum = RNS.Reticulum(
            configdir=self.config.rns_config_dir,
        )

        # Load or create identity
        if os.path.isfile(self.identity_path):
            self.identity = RNS.Identity.from_file(self.identity_path)
            if self.identity is None:
                raise RuntimeError(f"Failed to load identity from {self.identity_path}")
            log.info("Loaded identity %s", self.identity.hexhash)
        else:
            self.identity = RNS.Identity()
            self.identity.to_file(self.identity_path)
            log.info("Created new identity %s", self.identity.hexhash)

        # Initialize LXMF router
        self.lxmf_router = LXMF.LXMRouter(
            identity=self.identity,
            storagepath=self.storage_path,
        )

        self.local_destination = self.lxmf_router.register_delivery_identity(
            self.identity,
            display_name=self.config.display_name,
        )

        self.lxmf_router.register_delivery_callback(self._on_message_received)

        # Register announce handler
        self._announce_handler = AnnounceHandler(self)
        RNS.Transport.register_announce_handler(self._announce_handler)

        # Announce at start if configured
        if self.config.announce_at_start:
            self.announce()

        log.info("fieldmsg core initialized — %s", self.identity.hexhash)

    def announce(self) -> None:
        """Announce this destination on the network."""
        if self.local_destination:
            self.lxmf_router.announce(self.local_destination.hash)

    def send_message(self, dest_hash_hex: str, content: str) -> str | None:
        """Send an LXMF message to a destination hash. Returns message ID hex or None."""
        try:
            dest_hash = bytes.fromhex(dest_hash_hex)
            dest_identity = RNS.Identity.recall(dest_hash)

            if dest_identity is None:
                # Try to resolve via path request
                RNS.Transport.request_path(dest_hash)
                # For now, try sending without identity (propagated)
                lxm = LXMF.LXMessage(
                    destination=None,
                    source=self.local_destination,
                    content=content,
                    desired_method=LXMF.LXMessage.PROPAGATED,
                    destination_hash=dest_hash,
                )
            else:
                dest = RNS.Destination(
                    dest_identity,
                    RNS.Destination.OUT,
                    RNS.Destination.SINGLE,
                    "lxmf",
                    "delivery",
                )
                lxm = LXMF.LXMessage(
                    destination=dest,
                    source=self.local_destination,
                    content=content,
                    desired_method=LXMF.LXMessage.DIRECT,
                )

            msg_id = RNS.hexrep(lxm.hash, delimit=False)

            lxm.register_delivery_callback(self._on_delivery_status)
            lxm.register_failed_callback(self._on_delivery_status)

            self.lxmf_router.handle_outbound(lxm)

            # Store outbound message
            self.store.save_message(
                msg_id, dest_hash_hex, "out", content, time.time(), "pending",
            )

            return msg_id

        except Exception:
            log.exception("Failed to send message to %s", dest_hash_hex)
            return None

    def sync_propagation_node(self) -> None:
        """Request messages from configured propagation node."""
        prop_hash = self.config.propagation_node
        if not prop_hash:
            return
        try:
            prop_bytes = bytes.fromhex(prop_hash)
            self.lxmf_router.set_outbound_propagation_node(prop_bytes)
            self.lxmf_router.request_messages_from_propagation_node(self.identity)
            log.info("Requested messages from propagation node %s", prop_hash)
        except Exception:
            log.exception("Failed to sync propagation node")

    def _on_message_received(self, message: LXMF.LXMessage) -> None:
        """LXMF delivery callback — runs in Reticulum's thread."""
        try:
            source_hash = RNS.hexrep(message.source_hash, delimit=False)
            content = message.content_as_string() or ""
            msg_id = RNS.hexrep(message.hash, delimit=False)
            timestamp = message.timestamp if hasattr(message, "timestamp") else time.time()

            self.store.save_message(
                msg_id, source_hash, "in", content, timestamp, "delivered",
            )

            # Update contact display name from source if available
            display_name = None
            if hasattr(message, "source") and message.source:
                # Try to extract display name from cached announce data
                pass

            if self.on_message:
                self.on_message(msg_id, source_hash, content, timestamp)

        except Exception:
            log.exception("Error processing received message")

    def _on_delivery_status(self, message: LXMF.LXMessage) -> None:
        """Delivery status callback — runs in Reticulum's thread."""
        try:
            msg_id = RNS.hexrep(message.hash, delimit=False)
            state = message.state

            if state == LXMF.LXMessage.DELIVERED:
                status = "delivered"
            elif state == LXMF.LXMessage.FAILED:
                status = "failed"
            elif state == LXMF.LXMessage.SENT:
                status = "sent"
            else:
                status = "pending"

            self.store.update_message_status(msg_id, status)

            if self.on_delivery_status:
                self.on_delivery_status(msg_id, status)

        except Exception:
            log.exception("Error processing delivery status")

    def get_own_hash(self) -> str:
        """Return this node's destination hash as hex."""
        if self.local_destination:
            return RNS.hexrep(self.local_destination.hash, delimit=False)
        return self.identity.hexhash

    def shutdown(self) -> None:
        """Clean shutdown."""
        if self.store:
            self.store.close()
        if self.reticulum:
            self.reticulum.exit_handler()
            self.reticulum = None
```

```python
import os

def build_parser() -> argparse.ArgumentParser:
    parser = argparse.ArgumentParser(
        prog="fieldmsg",
        description="Lightweight LXMF messenger for Reticulum",
    )
    parser.add_argument("--config", metavar="PATH", help="path to fieldmsg config file")
    parser.add_argument("--rnsconfig", metavar="PATH", help="path to Reticulum config directory")
    parser.add_argument("--daemon", action="store_true", help="run headless without TUI")
    parser.add_argument("--generate-config", action="store_true", help="print example config to stdout and exit")

    sub = parser.add_subparsers(dest="command")
    send_p = sub.add_parser("send", help="send a message and exit")
    send_p.add_argument("destination", help="destination hash (hex)")
    send_p.add_argument("message", help="message text")

    return parser


def run_send(args, config) -> int:
    """Send a single message and exit."""
    from fieldmsg.core import Core

    core = Core(config)
    core.setup()

    delivered = None

    def on_status(msg_id, status):
        nonlocal delivered
        delivered = status

    core.on_delivery_status = on_status
    msg_id = core.send_message(args.destination, args.message)

    if msg_id is None:
        print("Failed to create message", file=sys.stderr)
        core.shutdown()
        return 1

    print(f"Sending {msg_id[:16]}...")

    # Wait up to 30 seconds for delivery status
    deadline = time.time() + 30
    while time.time() < deadline and delivered is None:
        time.sleep(0.5)

    if delivered == "delivered":
        print("Delivered.")
    elif delivered == "sent":
        print("Sent (delivery not confirmed yet).")
    elif delivered == "failed":
        print("Delivery failed.", file=sys.stderr)
        core.shutdown()
        return 1
    else:
        print("Timed out waiting for delivery status.")

    core.shutdown()
    return 0


def run_daemon(config) -> int:
    """Run headless LXMF router."""
    from fieldmsg.core import Core

    core = Core(config)
    core.setup()

    running = True

    def handle_signal(sig, frame):
        nonlocal running
        running = False

    signal.signal(signal.SIGINT, handle_signal)
    signal.signal(signal.SIGTERM, handle_signal)

    print(f"fieldmsg daemon running — {core.get_own_hash()}")

    last_sync = 0
    while running:
        time.sleep(1)
        # Periodic propagation sync
        if config.propagation_node and config.sync_interval > 0:
            if time.time() - last_sync >= config.sync_interval:
                core.sync_propagation_node()
                last_sync = time.time()

    print("Shutting down...")
    core.shutdown()
    return 0


def run_tui(config) -> int:
    """Launch the Textual TUI."""
    from fieldmsg.tui.app import FieldMsgApp

    app = FieldMsgApp(config)
    app.run()
    return 0


def main() -> None:
    parser = build_parser()
    args = parser.parse_args()

    if args.generate_config:
        from fieldmsg.config import generate_example_config
        print(generate_example_config())
        sys.exit(0)

    from fieldmsg.config import load_config
    config = load_config(args.config)
    config.rns_config_dir = args.rnsconfig

    if args.command == "send":
        sys.exit(run_send(args, config))
    elif args.daemon:
        sys.exit(run_daemon(config))
    else:
        sys.exit(run_tui(config))


if __name__ == "__main__":
    main()
```

```python
from __future__ import annotations

import time


def format_announce(hash: str, display_name: str | None, hops: int, timestamp: float, interface: str | None) -> str:
    """Format an announce entry for display."""
    name = display_name or hash[:16]
    age = _relative_time(timestamp)
    hop_str = f"{hops} hop{'s' if hops != 1 else ''}"
    iface = f" via {interface}" if interface else ""
    return f"{name} [{hash[:12]}..] {hop_str}{iface} ({age})"


def _relative_time(ts: float) -> str:
    delta = int(time.time() - ts)
    if delta < 60:
        return "just now"
    if delta < 3600:
        m = delta // 60
        return f"{m}m ago"
    if delta < 86400:
        h = delta // 3600
        return f"{h}h ago"
    d = delta // 86400
    return f"{d}d ago"
```

```python
    }

    ChatView .chat-title {
        text-style: bold;
        padding: 1;
        color: $accent;
        dock: top;
        height: 3;
    }

    ChatView #message-scroll {
        height: 1fr;
    }

    ChatView #compose-input {
        dock: bottom;
        margin: 0 1;
    }
    """

    def __init__(self, core: Core, peer_hash: str):
        super().__init__()
        self.core = core
        self.peer_hash = peer_hash

    def compose(self) -> ComposeResult:
        contact = self.core.store.get_contact(self.peer_hash)
        name = contact["nickname"] if contact else self.peer_hash[:16]
        yield Static(f"Chat with {name}", classes="chat-title")
        yield ScrollableContainer(id="message-scroll")
        yield Input(placeholder="Type a message...", id="compose-input")

    def on_mount(self) -> None:
        self.core.store.mark_read(self.peer_hash)
        self._load_messages()
        self.query_one("#compose-input", Input).focus()

        # Wire up live updates
        self.core.on_message = self._on_new_message
        self.core.on_delivery_status = self._on_status_update

    def _load_messages(self) -> None:
        scroll = self.query_one("#message-scroll", ScrollableContainer)
        messages = self.core.store.get_messages(self.peer_hash)
        for msg in messages:
            scroll.mount(MessageBubble(
                direction=msg["direction"],
                content=msg["content"],
                timestamp=msg["timestamp"],
                status=msg["status"],
            ))
        # Scroll to bottom
        scroll.scroll_end(animate=False)

    def _on_new_message(self, msg_id, source_hash, content, timestamp):
        if source_hash == self.peer_hash:
            self.app.call_from_thread(self._append_message, "in", content, timestamp, "delivered")
            self.app.call_from_thread(lambda: self.core.store.mark_read(self.peer_hash))

    def _on_status_update(self, msg_id, status):
        # Refresh messages to show updated status
        self.app.call_from_thread(self._refresh_messages)

    def _append_message(self, direction, content, timestamp, status):
        scroll = self.query_one("#message-scroll", ScrollableContainer)
        scroll.mount(MessageBubble(direction=direction, content=content, timestamp=timestamp, status=status))
        scroll.scroll_end(animate=False)

    def _refresh_messages(self):
        scroll = self.query_one("#message-scroll", ScrollableContainer)
        scroll.remove_children()
        self._load_messages()

    def on_input_submitted(self, event: Input.Submitted) -> None:
        text = event.value.strip()
        if not text:
            return
        event.input.value = ""

        msg_id = self.core.send_message(self.peer_hash, text)
        if msg_id:
            self._append_message("out", text, time.time(), "pending")
```
